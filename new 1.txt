  public class ReverseString{
  
     public static void main(String[]args)
	 {
	   String str = "Kishor";
	   String strRev = " ";
	   char ch;
	   
	   for(int i=str.length()-1;i>0;i--)
	   {
	       ch = str.charAt(i)
		   shrRev = strRev+ch;
	   }
	   System.out.println("Reverse String = " + strRev)
 }
 }
